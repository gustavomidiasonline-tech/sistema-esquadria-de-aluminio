## Schema

```sql
-- Ordens de producao
CREATE TABLE production_orders (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  company_id UUID NOT NULL REFERENCES companies(id),
  pedido_id UUID NOT NULL REFERENCES pedidos(id),
  numero TEXT NOT NULL,          -- OP-2026-001
  status TEXT NOT NULL DEFAULT 'pendente'
    CHECK (status IN ('pendente','aguardando_material','em_producao','qualidade','pronto','entregue','cancelada')),
  data_inicio_prevista DATE,
  data_entrega_prevista DATE,
  data_inicio_real TIMESTAMPTZ,
  data_conclusao_real TIMESTAMPTZ,
  responsavel_id UUID REFERENCES profiles(id),
  observacoes TEXT,
  plano_corte_gerado BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Itens da ordem de producao (pecas individuais)
CREATE TABLE production_order_items (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  production_order_id UUID NOT NULL REFERENCES production_orders(id),
  pedido_item_id UUID REFERENCES pedido_itens(id),
  perfil_aluminio_id UUID REFERENCES perfis_aluminio(id),
  posicao TEXT,
  comprimento_mm NUMERIC NOT NULL,
  quantidade INTEGER NOT NULL DEFAULT 1,
  status TEXT DEFAULT 'pendente',
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Lista de materiais calculada
CREATE TABLE materials_list (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  company_id UUID NOT NULL REFERENCES companies(id),
  production_order_id UUID REFERENCES production_orders(id),
  orcamento_id UUID REFERENCES orcamentos(id),
  perfil_aluminio_id UUID REFERENCES perfis_aluminio(id),
  glass_type_id UUID REFERENCES glass_types(id),
  hardware_id UUID REFERENCES hardware(id),
  descricao TEXT,
  quantidade NUMERIC NOT NULL,
  unidade TEXT DEFAULT 'un',
  custo_unitario NUMERIC,
  custo_total NUMERIC GENERATED ALWAYS AS (quantidade * custo_unitario) STORED,
  created_at TIMESTAMPTZ DEFAULT now()
);
```
